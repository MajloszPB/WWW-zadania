Jak programować w JavaScript

Tworzenie klasy w JavaScript


class Nazwa {
    constructor(atrybuty) {
        this.atrybuty = atrybuty;
    }

    metoda() {
        return this.atrybuty;
    }
}

Funkcja główna (main)


function main() {
    const obiekt = new Nazwa(10);
    console.log(obiekt.metoda());
}

main();

Jak programować w C#

Tworzenie klasy w C#


class Nazwa {
    public int atrybuty;

    public Nazwa(int atrybuty) {
        this.atrybuty = atrybuty;
    }

    public int Metoda() {
        return atrybuty;
    }
}

Metoda Main


class Program {
    public static void Main(string[] args) {
        Nazwa obiekt = new Nazwa(5);
        Console.WriteLine(obiekt.Metoda());
    }
}

Jak programować w Pythonie

Tworzenie klasy w Pythonie


class Nazwa:
    def __init__(self, atrybuty):
        self.atrybuty = atrybuty

    def metoda(self):
        return self.atrybuty

Funkcja main


def main():
    obiekt = Nazwa(10)
    print(obiekt.metoda())

if __name__ == "__main__":
    main()
